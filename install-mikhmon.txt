#bin/bash
pkg install git php -y
git clone https://github.com/mihawk13/mikhmonv3.git
cat >start-mikhmonv3 <<EOL
#bin/bash
if pgrep "php" >/dev/null 2>&1 ; then
killall php
fi
cd mikhmonv3
php -S 0.0.0.0:8080
EOL
cat >update-mikhmonv3 <<EOL
#bin/bash
rm -rf ~/backup-mikhmonv3
mkdir ~/backup-mikhmonv3
cp ~/mikhmonv3/include/config.php ~/backup-mikhmonv3/config.php
cp -rf ~/mikhmonv3/voucher/. ~/backup-mikhmonv3/voucher/
cp -rf ~/mikhmonv3/img/. ~/backup-mikhmonv3/img/
rm -rf mikhmonv3
rm ~/backup-mikhmonv3/voucher/print.php
rm ~/backup-mikhmonv3/voucher/printbt.php
git clone https://github.com/mihawk13/mikhmonv3.git
cp ~/backup-mikhmonv3/config.php ~/mikhmonv3/include/config.php
cp -rf ~/backup-mikhmonv3/voucher/. ~/mikhmonv3/voucher/
cp -rf ~/backup-mikhmonv3/img/. ~/mikhmonv3/img/
rm -rf ~/backup-mikhmonv3
./start-mikhmonv3
EOL
cat >remove-mikhmonv3 <<EOL
#bin/bash
rm -rf mikhmonv3
EOL
cat >.bashrc <<EOL
~/start-mikhmonv3
EOL
chmod +x start-mikhmonv3
chmod +x update-mikhmonv3
chmod +x remove-mikhmonv3
./start-mikhmonv3
